<CodeGroup tabs="JavaScript,Python">
  ```javascript
  const nxtjob = require('nxtjob-sdk');
  const client = new nxtjob.Client({ apiKey: 'YOUR_API_KEY' });
  
  async function trackJob(jobId) {
    const status = await client.jobs.track(jobId, { stage: 'applied' });
    console.log(status);
  }
  ```
